    2. Utworz src/types/ai.ts z podstawowymi typami:
       ```typescript
       import { z } from 'zod';

       // Task types for model routing
       export type AITask = 'mentor' | 'curriculum' | 'quiz' | 'embedding';

       // Cost tracking
       export interface CostLog {
         task: AITask;
         model: string;
         inputTokens: number;
         outputTokens: number;
         timestamp: Date;
         durationMs?: number;
       }

       // Model configuration
       export interface ModelConfig {
         mentor: string;
         curriculum: string;
         quiz: string;
         embedding: string;
       }

       // Curriculum schemas (will be expanded in Phase 2)
       export const ChapterSchema = z.object({
         title: z.string(),
         description: z.string(),
         estimatedTime: z.string(),
         topics: z.array(z.string()),
       });

       export const LevelSchema = z.object({
         level: z.number().min(1).max(5),
         name: z.enum(['Poczatkujacy', 'Podstawowy', 'Sredniozaawansowany', 'Zaawansowany', 'Guru']),
         chapters: z.array(ChapterSchema),
         learningOutcomes: z.array(z.string()),
       });

       export const CurriculumSchema = z.object({
         title: z.string(),
         topic: z.string(),
         levels: z.array(LevelSchema),
       });

       export type Chapter = z.infer<typeof ChapterSchema>;
       export type Level = z.infer<typeof LevelSchema>;
       export type Curriculum = z.infer<typeof CurriculumSchema>;
       ```
