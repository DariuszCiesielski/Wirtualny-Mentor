  1. Stworz migracje SQL `supabase/migrations/20260309000001_business_suggestions.sql`:

  ```sql
  -- Phase 9: Business Suggestions

  -- Add profile_version to user_business_profiles (for cache invalidation)
  ALTER TABLE user_business_profiles
    ADD COLUMN profile_version INT NOT NULL DEFAULT 1;

  -- Business suggestions table
  CREATE TABLE business_suggestions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
    course_id UUID NOT NULL REFERENCES courses(id) ON DELETE CASCADE,
    chapter_id UUID NOT NULL REFERENCES chapters(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT NOT NULL,
    business_potential TEXT,
    estimated_complexity TEXT CHECK (estimated_complexity IN ('prosty', 'sredni', 'zlozony')),
    relevant_section TEXT,
    model_name TEXT,
    input_hash TEXT,
    profile_version INT,
    is_bookmarked BOOLEAN NOT NULL DEFAULT FALSE,
    is_dismissed BOOLEAN NOT NULL DEFAULT FALSE,
    dismissed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now()
  );

  -- Index for rate limit query (user + date)
  CREATE INDEX idx_business_suggestions_user_date
    ON business_suggestions (user_id, created_at DESC);

  -- Index for cache lookup (user + chapter)
  CREATE INDEX idx_business_suggestions_user_chapter
    ON business_suggestions (user_id, chapter_id, created_at DESC);

  -- RLS
  ALTER TABLE business_suggestions ENABLE ROW LEVEL SECURITY;

  CREATE POLICY "Users can view own suggestions"
    ON business_suggestions FOR SELECT
    USING (auth.uid() = user_id);

  CREATE POLICY "Users can insert own suggestions"
    ON business_suggestions FOR INSERT
    WITH CHECK (auth.uid() = user_id);

  CREATE POLICY "Users can update own suggestions"
    ON business_suggestions FOR UPDATE
    USING (auth.uid() = user_id);
  ```

  UWAGA: Uzyj `'sredni'` (bez polskiego znaku) w CHECK constraint — PostgRES moze miec problemy z diakrytykami w CHECK. Wartosc uzytkownikowi wyswietlana bedzie z mapowaniem w UI.
